Import-Module SqlServer -ErrorAction Stop

$ServerInstance = "your-mi-name.public.xxxxx.database.windows.net"
$DatabaseName   = "YourDatabase"
$Username       = "your_sql_user"
$Password       = "your_password"
$SqlFolder      = "C:\Automation\SQL"
$OutputFolder   = "C:\Automation\Output"

if (-not (Test-Path $OutputFolder)) {
    New-Item -Path $OutputFolder -ItemType Directory -Force | Out-Null
}

Get-ChildItem -Path $SqlFolder -Filter *.sql -File | ForEach-Object {
    $queryText = Get-Content $_.FullName -Raw
    $csvPath = Join-Path $OutputFolder "$($_.BaseName).csv"

    Invoke-Sqlcmd `
        -ServerInstance $ServerInstance `
        -Database $DatabaseName `
        -Username $Username `
        -Password $Password `
        -Query $queryText `
        -Encrypt Mandatory `
        -TrustServerCertificate |
    Export-Csv -Path $csvPath -NoTypeInformation -Encoding UTF8

    Write-Host "Exported $csvPath"
}
